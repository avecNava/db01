           e.city,
           c.company_name,
           c.city
FROM       employee e
INNER JOIN works w
ON         e.employee_name=w.employee_name
INNER JOIN company c
ON         w.company_name = c.company_name

-- trigger to check if employees and company are from same cityCREATE TRIGGER trigger_emp_company_from_same_city before
INSERT
OR
UPDATE
ON PUBLIC.works FOR each row
EXECUTE PROCEDURE
  PUBLIC.fn_emp_company_from_same_city();
-- trgger function
create
OR
replace FUNCTION PUBLIC.fn_emp_company_from_same_city()
returns TRIGGER AS $$
BEGIN
  IF EXISTS
  (
             SELECT
             FROM       employee e
             INNER JOIN company c
             ON         c.city=e.city
             WHERE      e.employee_name=new.employee_name
             AND        c.company_name=new.company_name ) THEN
  raise notice '** Employee: % and Company : % are from same city.', new.employee_name, new.company_name;
  return new;
  else raise notice '** Employee: % and Company : % are from different city.', new.employee_name, new.company_name;
end
IF;
return NULL;
end;
$$ language 'plpgsql';
